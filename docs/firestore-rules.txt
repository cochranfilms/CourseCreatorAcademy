rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    // Helper function to check if user is authenticated
    function isAuthenticated() {
      return request.auth != null;
    }
    
    // Pending membership claims:
    // Created by server-side webhook when buyer email is known but no userId is present.
    // Allow the authenticated user whose email matches to mark the record as claimed.
    match /pendingMemberships/{claimId} {
      function authed() { return request.auth != null; }
      function emailMatches() {
        // Safely compare lower-cased emails
        return authed()
          && request.auth.token.email != null
          && lower(resource.data.email) == lower(request.auth.token.email);
      }
      allow read: if emailMatches();
      // Restrict updates to only marking as claimed by the current user
      allow update: if emailMatches()
        && request.resource.data.diff(resource.data).changedKeys().hasOnly(['claimed','claimedBy','claimedAt'])
        && request.resource.data.claimed == true
        && request.resource.data.claimedBy == request.auth.uid;
      // Creation/deletion happens server-side only
      allow create, delete: if false;
    }

    // Legacy subscriptions - allow users to read their own subscription docs
    match /legacySubscriptions/{subId} {
      allow read: if isAuthenticated() && resource.data.userId == request.auth.uid;
      // Writes are performed by server webhooks only
      allow write: if false;
    }

    // Legacy creators catalog (public)
    match /legacy_creators/{creatorId} {
      allow read: if true;
      // Subcollections: public sample videos
      match /videos/{videoId} {
        allow read: if true;
      }
    }

    // Helper function to check if user owns the resource
    function isOwner(userId) {
      return isAuthenticated() && request.auth.uid == userId;
    }
    
    // Latest-terms helper (reads config doc)
    function hasAcceptedLatestTerms(userId) {
      let user = get(/databases/$(database)/documents/users/$(userId)).data;
      let config = get(/databases/$(database)/documents/config/terms).data;
      return user.terms.accepted == true && user.terms.version == config.requiredVersion;
    }
    
    // Users collection - users can read/write their own data
    // Public profiles: safe subset exposed for public reads
    match /publicProfiles/{userId} {
      allow read: if true;
      allow write: if isAuthenticated() && isOwner(userId);
    }
    
    match /users/{userId} {
      // Allow authenticated users to read profile docs so avatars/names can be shown across the app.
      // Note: Keep sensitive data out of this document; store private data in subcollections.
      allow read: if isAuthenticated();
      allow create: if isAuthenticated() && isOwner(userId);
      allow update: if isAuthenticated() && isOwner(userId);
      allow delete: if isAuthenticated() && isOwner(userId);
      
      // Saved items subcollection - users can only access their own saved items
      match /saved/{savedId} {
        allow read, write: if isAuthenticated() && isOwner(userId);
      }
      
      // Progress tracking subcollection - users can only access their own progress
      match /progress/{progressId} {
        allow read, write: if isAuthenticated() && isOwner(userId);
        
        // Courses progress subcollection
        match /courses/{courseId} {
          allow read, write: if isAuthenticated() && isOwner(userId);
        }
      }
    }
    
    // Orders (Marketplace)
    match /orders/{orderId} {
      function isAuthed() { return request.auth != null; }
      function isBuyer() { return isAuthed() && request.auth.uid == resource.data.buyerId; }
      function isSeller() { return isAuthed() && request.auth.uid == resource.data.sellerId; }

      // Read allowed for buyer or seller
      allow read: if isBuyer() || isSeller();

      // Create via server/webhook only
      allow create: if false;

      // Updates:
      // - Seller can set tracking and mark shipped
      // - Buyer can mark delivered
      allow update: if (
        (isSeller() &&
          request.resource.data.diff(resource.data).changedKeys().hasOnly(
            ['trackingNumber','trackingCarrier','trackingUrl','status','shippedAt']
          ) && request.resource.data.status in ['awaiting_tracking','shipped']
        ) ||
        (isBuyer() &&
          request.resource.data.diff(resource.data).changedKeys().hasOnly(['status','deliveredAt']) &&
          request.resource.data.status == 'delivered'
        )
      );

      // Delete blocked
      allow delete: if false;
    }

    // Marketplace listings collection
    match /listings/{listingId} {
      // Anyone authenticated can read listings
      allow read: if isAuthenticated();
      
      // Only authenticated users can create listings
      // Creator ID must match the authenticated user
      allow create: if isAuthenticated() 
        && request.resource.data.creatorId == request.auth.uid
        && request.resource.data.keys().hasAll(['title', 'price', 'condition', 'creatorId', 'createdAt'])
        && hasAcceptedLatestTerms(request.auth.uid);
      
      // Only the creator can update/delete their listing
      allow update: if isAuthenticated() 
        && resource.data.creatorId == request.auth.uid
        && request.resource.data.creatorId == request.auth.uid;
      
      allow delete: if isAuthenticated() && resource.data.creatorId == request.auth.uid;
    }
    
    // Opportunities collection
    match /opportunities/{opportunityId} {
      // Anyone authenticated can read opportunities
      allow read: if isAuthenticated();
      
      // Only authenticated users can create opportunities
      // Poster ID must match the authenticated user
      allow create: if isAuthenticated()
        && request.resource.data.posterId == request.auth.uid
        && request.resource.data.keys().hasAll(['title', 'company', 'location', 'type', 'posterId', 'posted']);
      
      // Only the poster can update/delete their opportunity
      allow update: if isAuthenticated()
        && resource.data.posterId == request.auth.uid
        && request.resource.data.posterId == request.auth.uid;
      
      allow delete: if isAuthenticated() && resource.data.posterId == request.auth.uid;
    }
    
    // Job Applications collection
    match /jobApplications/{applicationId} {
      function isApplicant() {
        return isAuthenticated() && request.auth.uid == resource.data.applicantId;
      }
      function isPoster() {
        return isAuthenticated() && request.auth.uid == resource.data.posterId;
      }
      
      // Users can read applications where they are the applicant or poster
      // This works for both single document reads and queries
      // Firestore evaluates each document in a query against these rules
      allow read: if isAuthenticated() && (isApplicant() || isPoster());
      
      // Only authenticated users can create applications
      // Applicant ID must match the authenticated user
      // Cannot apply to own opportunity
      allow create: if isAuthenticated()
        && request.resource.data.applicantId == request.auth.uid
        && request.resource.data.keys().hasAll(['opportunityId', 'applicantId', 'posterId', 'name', 'email', 'coverLetter', 'status', 'createdAt'])
        && request.resource.data.status == 'pending';
      
      // Updates:
      // - Poster can hire (status: pending -> hired)
      // - Poster can reject (status: pending -> rejected)
      // - Applicant can mark complete (status: hired -> completed)
      // - Server can update payment fields (via admin)
      allow update: if isAuthenticated() && (
        // Poster can hire or reject
        (isPoster() && 
          resource.data.status == 'pending' &&
          request.resource.data.status in ['hired', 'rejected'] &&
          request.resource.data.applicantId == resource.data.applicantId &&
          request.resource.data.posterId == resource.data.posterId) ||
        // Applicant can mark complete
        (isApplicant() &&
          resource.data.status == 'hired' &&
          request.resource.data.status == 'completed' &&
          request.resource.data.applicantId == resource.data.applicantId &&
          request.resource.data.posterId == resource.data.posterId)
      );
      
      // Deletion not allowed (for audit trail)
      allow delete: if false;
    }
    
    // Projects collection
    match /projects/{projectId} {
      // Anyone authenticated can read projects
      allow read: if isAuthenticated();
      
      // Only authenticated users can create projects
      // Creator ID must match the authenticated user
      allow create: if isAuthenticated()
        && request.resource.data.creatorId == request.auth.uid;
      
      // Only the creator can update/delete their project
      allow update: if isAuthenticated()
        && resource.data.creatorId == request.auth.uid;
      
      allow delete: if isAuthenticated() && resource.data.creatorId == request.auth.uid;
    }
    
    // Threads collection (for messaging)
    match /threads/{threadId} {
      // Users can read threads they are members of
      allow read: if isAuthenticated() && request.auth.uid in resource.data.members;
      
      // Users can create threads they are members of
      allow create: if isAuthenticated() 
        && request.auth.uid in request.resource.data.members;
      
      // Users can update threads they are members of
      allow update: if isAuthenticated() 
        && request.auth.uid in resource.data.members;
      
      // Users can delete threads they are members of (optional - you may want to restrict this)
      allow delete: if isAuthenticated() && request.auth.uid in resource.data.members;
      
      // Messages subcollection
      match /messages/{messageId} {
        // Users can read messages in threads they are members of
        allow read: if isAuthenticated() 
          && request.auth.uid in get(/databases/$(database)/documents/threads/$(threadId)).data.members;
        
        // Users can create messages in threads they are members of
        allow create: if isAuthenticated() 
          && request.auth.uid == request.resource.data.senderId
          && request.auth.uid in get(/databases/$(database)/documents/threads/$(threadId)).data.members;
        
        // Users can update messages they sent (for read receipts, etc.)
        allow update: if isAuthenticated() 
          && request.auth.uid in get(/databases/$(database)/documents/threads/$(threadId)).data.members;
        
        // Users can delete their own messages
        allow delete: if isAuthenticated() 
          && resource.data.senderId == request.auth.uid;
      }
    }
    
    // Courses (public catalog metadata; writes restricted)
    match /courses/{courseId} {
      // anyone can read course info
      allow read: if true;

      // only authenticated users can create courses
      allow create: if isAuthenticated();

      // only the course creator can update/delete the course
      allow update, delete: if isAuthenticated()
        && resource.data.creatorId == request.auth.uid;

      // Modules under a course
      match /modules/{moduleId} {
        // public can read module titles/order
        allow read: if true;

        // writes allowed only if caller is the course creator
        allow create, update, delete: if isAuthenticated()
          && get(/databases/$(database)/documents/courses/$(courseId)).data.creatorId == request.auth.uid;

        // Lessons under a module
        match /lessons/{lessonId} {
          // public can read lesson titles/order/duration
          allow read: if true;

          // writes allowed only if caller is the course creator
          allow create, update, delete: if isAuthenticated()
            && get(/databases/$(database)/documents/courses/$(courseId)).data.creatorId == request.auth.uid;
        }
      }
    }
    
    // Feedback collection - test users can submit feedback
    match /feedback/{feedbackId} {
      // Users can read their own feedback submissions
      allow read: if isAuthenticated() && resource.data.userId == request.auth.uid;
      
      // Authenticated users can create feedback submissions
      // User ID must match the authenticated user
      allow create: if isAuthenticated()
        && request.resource.data.userId == request.auth.uid
        && request.resource.data.keys().hasAll(['userId', 'feedback', 'createdAt', 'status'])
        && request.resource.data.status == 'new';
      
      // Users can update their own feedback (e.g., to add more details)
      allow update: if isAuthenticated()
        && resource.data.userId == request.auth.uid
        && request.resource.data.userId == request.auth.uid;
      
      // Deletion not allowed (for audit trail)
      allow delete: if false;
    }
    
    // Assets collection - premium downloadable assets (LUTs, presets, SFX, templates, etc.)
    match /assets/{assetId} {
      // Allow authenticated users to read assets
      allow read: if isAuthenticated();
      
      // Writes are performed by server/admin only (via Firebase Admin SDK)
      allow write: if false;
    }
    
    // Deny all other access
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
